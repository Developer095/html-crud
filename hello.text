    <h3 id="loading"></h3>
    <div id="products"></div>

</body>

<script>
    const load = document.getElementById("loading");
    const prod = document.getElementById("products");
    // load.textContent = "Loading...";
    document.addEventListener("DOMContentLoaded", async () => {
        load.textContent = "Loading...";

        try {
            const response = await fetch('https://api.onlinecrockerystore.com/tasks');

            if (!response.ok) {
                const error = await response.json();
                throw new Error(error);
            }
            const category = await response.json();
            console.log("Categories Data", category);
            const categoryHtml = category.map((item, idx) => {
                return `<div>ID: ${item.id}</div>
                <div>Name: ${item.name} </div>
                `;
            }).join("");
            prod.innerHTML = categoryHtml;
            load.textContent = "";

        } catch (error) {
            console.log("Error Occured");
            load.textContent = "Error Occured";
            load.style.background = "red";
            load.style.color = "white";

        }
    })

</script>

</html>
